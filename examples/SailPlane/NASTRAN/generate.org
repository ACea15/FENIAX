  def wb2w(point_le, x_te, z_bottom, pct2le, pct2te):
      """ Wing-box to wing"""

      x = x_te - point_le[0]
      pct_wb = (pct2te - pct2le)
      chord = x / pct_wb
      LE = point_le[0] - chord * pct2le
      point_LE = [LE, point_le[1], (point_le[2] +
                                    z_bottom) / 2]
      return point_LE, chord
